=head1 NAME

Trocla - Interact with trocla for password management.

=head1 SYNOPSIS

 use Trocla;
   
 task "prepare", "host01", sub {
   my $trocla = Trocla->new(host => "trocla-server");
   my $db_pw = $trocla->get("db-password-" . connection->server);
 };
